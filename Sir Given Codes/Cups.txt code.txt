#include<iostream>
#include<string>
using namespace std;

class Cups
{
    private:
        string cupType;
        double cupCapacity;

    public:
        Cups(string cupT, double cupC)
        {
            cupType = cupT;
            cupCapacity = cupC;
        }

        string getCupType()
        {
            return cupType;
        }

        void setCupType(string cupT)
        {
            cupType = cupT;
        }

        void Display()
        {
        cout << "This Cup type is " << cupType << " and Capacity is " << cupCapacity << "ml." <<endl;
        }

        void drink(){}

        void eat(){}
};


class Tea : public Cups
{
    public:
        Tea(string ct, double cc) : Cups(ct, cc){};

        void drink()
        {
            cout << "slurp, or blow on hot tea and sip\n\n";
        }
};
